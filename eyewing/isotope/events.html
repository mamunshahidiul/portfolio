Events

Isotope is an Event Emitter. You can bind listeners to events with the on and off methods.

var $container = $('#container').isotope({...});
var iso = new Isotope( elem );

function onArrange() {
  console.log('layout done');
}
// bind event listener
$container.isotope( 'on', 'arrangeComplete', onArrange );
// un-bind event listener
$container.isotope( 'off', 'arrangeComplete', onArrange );
// bind event listener to be triggered just once
$container.isotope( 'once', 'arrangeComplete', function() {
  console.log('arrange done, just this one time');
});

arrangeComplete

$container.isotope( 'on', 'arrangeComplete', function( filteredItems ) {} )
// or with vanilla JS
iso.on( 'arrangeComplete', function( filteredItems ) {} )
// or with jQuery
  • filteredItems

    Type: Array of Isotope.Items

    items that are visible

Triggered after .isotope({...})/arrange, when all layout, hide, and reveal transitions have completeled.

$container.isotope( 'on', 'arrangeComplete',
  function( filteredItems ) {
    console.log( 'Isotope arrange completed on ' +
      filteredItems.length + ' items' );
  }
);

layoutComplete

$container.isotope( 'on', 'layoutComplete', function( laidOutItems ) {} )
// or with vanilla JS
iso.on( 'layoutComplete', function( laidOutItems ) {} )
// or with jQuery
  • laidOutItems

    Type: Array of Isotope.Items

    items that were laid out

Triggered after a layout and all positioning transitions have been completed.

$container.isotope( 'on', 'layoutComplete',
  function( laidOutItems ) {
    console.log( 'Isotope layout completed on ' +
      laidOutItems.length + ' items' );
  }
);
